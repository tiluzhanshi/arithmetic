function maopao(a) {
  let len = a.length;
  for(let i = 1; i < len; i++) {
    let temp = a[i];
    
    for( var j = i; i > 0 && a[j-1] > temp; j--) {
      	a[j] = a[j - 1];
    }
    a[j] = temp;
  }
  return a;
}

function computerInfo(arr) {
  console.time("s")
  var arr = maopao(arr);
  console.timeEnd("s");
  return arr;
}
